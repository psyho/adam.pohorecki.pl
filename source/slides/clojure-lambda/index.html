class IfNot
  def then(&block)
    @then = block
  end

  def else(&block)
    @else = block
  end

  def run(condition)
    if !condition
      @then.call if @then
    else
      @else.call if @else
    end
  end
end

def if_not(condition, &block)
  runner = IfNot.new
  runner.instance_eval(&block)
  runner.run(condition)
end
