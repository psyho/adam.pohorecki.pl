var datepicker = function() {
  return {
    require: 'ngModel',
    link: function(scope, el, attrs, ctrl) {
      jQuery(el).datepicker({
        onSelect: function(dateText) {
          scope.$apply(function() {
            ctrl.$setViewValue(dateText);
          });
        }
      });

      ctrl.render = function() {
        jQuery(el).datepicker('setDate',
          ctrl.$viewValue);
      };
    }
  };
};

  directive('datepicker', datepicker);

What sucks in Angular

A year ago:

  • no forms
  • non-composable validations
  • no i18n
  • complex directives
  • directives/widgets
  • scope and controller were the same thing

All my complaints were addressed

Plus stuff I didn't know I needed

Big improvements => lots of breaking changes

1.0 should be stable though

Problems now:

  • animations require more work
  • $resource could be similar to backbone models
  • very few libraries extending Angular
  • can't reuse templates on the backend

Highlights

Easy setup

Unmatched testability

Cleaner code with less callbacks an boilerplate
