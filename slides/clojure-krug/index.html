(defn in-transaction' [block]
  (try
    (let [t (start-transaction)]
      (block)
      (commit t))
    (catch Exception e (rollback t))))

(defn feed [animal food]
  (if (can-eat? animal food)
    (eat animal food)
    (throw (Exception. "BARF!")))

(defprotocol Eater
  (can-eat? [animal food])
  (eat [animal food]))

(defn herbivore [_ food]
  (= :vegetable (:food-type food)))
